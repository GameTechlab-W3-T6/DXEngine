# 텍스트는 저장소에는 LF로 정규화, OS별 체크아웃은 자동
* text=auto

# 코드/셰이더는 LF 고정
*.cpp text eol=lf
*.h   text eol=lf
*.hlsl text eol=lf
*.glsl text eol=lf
*.py  text eol=lf
*.sh  text eol=lf

# VS 관련 파일은 CRLF가 편함
*.sln     text eol=crlf
*.vcxproj text eol=crlf
*.props   text eol=crlf
*.filters text eol=crlf
*.bat     text eol=crlf
